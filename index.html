// AES-128 Decryption Implementation
void aesDecrypt(uint8_t* ciphertext, 
                uint8_t* key, 
                uint8_t* plaintext) {
    
    // Key expansion for decryption
    uint8_t roundKeys[11][16];
    keyExpansion(key, roundKeys);
    
    // Initial round
    addRoundKey(ciphertext, roundKeys[10]);
    
    // Main rounds
    for (int round = 9; round > 0; round--) {
        invShiftRows(ciphertext);
        invSubBytes(ciphertext);
        addRoundKey(ciphertext, roundKeys[round]);
        invMixColumns(ciphertext);
    }
    
    // Final round
    invShiftRows(ciphertext);
    invSubBytes(ciphertext);
    addRoundKey(ciphertext, roundKeys[0]);
    
    // Copy result to plaintext
    memcpy(plaintext, ciphertext, 16);
}
